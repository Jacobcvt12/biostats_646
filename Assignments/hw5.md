    $$
    \begin{aligned}
    {n \choose r} &= \frac{n!}{r!(n-r)!} \\
    &= \frac{[r+(n-r)](n-1)!}{r!(n-r)!} \\
    &= \frac{(n-1)!r+(n-1)!(n-r)}{r!(n-r)!} \\
    &= \frac{(n-1)!r}{r!(n-r)!} + \frac{(n-r)(n-1)!}{r!(n-r)!} \\
    &= \boxed{{n-1 \choose r-1} + {n-1 \choose r}}
    \end{aligned}
    $$
